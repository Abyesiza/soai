To make this truly revolutionary, we add **Probabilistic Prefetching**.

While the user is still in "Superposition" (e.g., probability is 0.5), the site doesn't just sit there. It uses the `intent` state to talk to the **Predictive Shadow DOM**.

1. **Ghost Loading:** If `probability` moves to 0.6, the browser starts a low-priority fetch for the "Data-Dense" component.
2. **The Zero-Frame Swap:** By the time the user's intent hits 0.8 (The Collapse), the component is already compiled and sitting in memory. The transition is 0ms because it’s just an `opacity` and `pointer-events` swap.
3. **Agentic Refinement:** We can pass this `intent` vector to a background Gemini process. The AI can then prepare a custom "Greeting" or "Offer" that matches the user's detected psychological state (e.g., "You seem to be looking for technical specs—here is a PDF comparison sheet").

import { atom, useAtom } from 'jotai';
import { useEffect, useRef } from 'react';

/**
 * ATOMIC STATE: The 'Superposition' Probability
 * 0.0 = Neutral
 * > 0.7 = Collapse into 'Transactional/Data' mode
 * < 0.3 = Collapse into 'Storytelling/Brand' mode
 */
export const intentAtom = atom({
  probability: 0.5,
  persona: 'neutral',
  velocity: 0,
});

export function useIntent() {
  const [intent, setIntent] = useAtom(intentAtom);
  const metrics = useRef({
    lastPos: { x: 0, y: 0 },
    lastTime: Date.now(),
    dwellTime: 0,
    scrollPositions: [],
  });

  useEffect(() => {
    const handleInteraction = (e) => {
      const now = Date.now();
      const dt = now - metrics.current.lastTime;
      
      // 1. Calculate Mouse Velocity
      const dx = e.clientX - metrics.current.lastPos.x;
      const dy = e.clientY - metrics.current.lastPos.y;
      const distance = Math.sqrt(dx * dx + dy * dy);
      const velocity = distance / dt;

      // 2. Identify 'Analytical' vs 'Emotional' Pathing
      // High Velocity + High Scroll = The 'Speedster' (wants CTA/Data)
      // Low Velocity + Long Dwell = The 'Researcher' (wants deep content)
      
      let newProb = intent.probability;

      if (velocity > 2.5) {
        // User is searching for something specific; lean toward Data/Direct
        newProb = Math.min(newProb + 0.05, 1.0);
      } else if (velocity < 0.2 && velocity > 0) {
        // User is lingering/reading; lean toward Story/Brand
        newProb = Math.max(newProb - 0.02, 0.0);
      }

      // Update the Atom (This triggers the 'Collapse' in the UI)
      setIntent({
        probability: newProb,
        persona: newProb > 0.7 ? 'analytical' : newProb < 0.3 ? 'storyteller' : 'neutral',
        velocity: velocity,
      });

      metrics.current.lastPos = { x: e.clientX, y: e.clientY };
      metrics.current.lastTime = now;
    };

    window.addEventListener('mousemove', handleInteraction);
    return () => window.removeEventListener('mousemove', handleInteraction);
  }, [intent.probability, setIntent]);

  return intent;
}
/**
 * COMPONENT: SuperpositionContainer
 * Handles the "Predictive Shadow DOM" logic.
 * It renders all potential states but keeps the "non-collapsed" ones in a ghost state (opacity-0).
 */
export function SuperpositionContainer({ children, analyticalView, storytellerView, neutralView }) {
  const intent = useIntent();
  const prob = intent.probability;

  // Define transition constants for smooth morphing
  const springConfig = { type: 'spring', stiffness: 300, damping: 30 };

  return (
    <div className="relative w-full min-h-[500px]">
      <AnimatePresence mode="popLayout">
        {/* ANALYTICAL COLLAPSE */}
        {prob > 0.7 && (
          <motion.div
            key="analytical"
            initial={{ opacity: 0, scale: 0.98 }}
            animate={{ opacity: 1, scale: 1 }}
            exit={{ opacity: 0, scale: 1.02 }}
            transition={springConfig}
            className="w-full"
          >
            {analyticalView}
          </motion.div>
        )}

        {/* STORYTELLER COLLAPSE */}
        {prob < 0.3 && (
          <motion.div
            key="storyteller"
            initial={{ opacity: 0, y: 20 }}
            animate={{ opacity: 1, y: 0 }}
            exit={{ opacity: 0, y: -20 }}
            transition={springConfig}
            className="w-full"
          >
            {storytellerView}
          </motion.div>
        )}

        {/* NEUTRAL / SUPERPOSITION STATE */}
        {prob >= 0.3 && prob <= 0.7 && (
          <motion.div
            key="neutral"
            initial={{ opacity: 0 }}
            animate={{ opacity: 1 }}
            exit={{ opacity: 0 }}
            className="w-full"
          >
            {neutralView || children}
          </motion.div>
        )}
      </AnimatePresence>

      {/* GHOST PRE-RENDERING (Hidden from view but in memory) */}
      <div className="pointer-events-none absolute inset-0 -z-50 opacity-0 overflow-hidden h-0">
        {/* We mount these to ensure assets/data are pre-fetched by the browser */}
        <div id="ghost-analytical">{analyticalView}</div>
        <div id="ghost-storyteller">{storytellerView}</div>
      </div>
    </div>
  );
}
